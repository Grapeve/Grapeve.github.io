def next_permutation(strs: list, use: set, path: str, ans: list):
    if len(use) == len(strs):
        ans.append(path)
    else:
        for i in range(len(strs)):
            if i not in use:
                use.add(i)
                next_permutation(strs, use, path + strs[i], ans)
                use.remove(i)

class Program:
    def __init__(self, profit, capital):
        self.profit = profit
        self.capital = capital

    def __lt__(self, program):
        return self.capital < program.capital


def find_max_capital(K: int, W: int, Profits: list[int], Capital: list[int]):
    programs = []
    for i in range(len(Profits)):
        programs.append(Program(Profits[i], Capital[i]))

    min_heap: list[Program] = []
    max_heap: list[Program] = []

    for program in programs:
        heapq.heappush(min_heap, program)  # å°æ ¹å †æ ¹æ®capital

    for i in range(K):
        while len(min_heap) != 0 and min_heap[0].capital <= W:
            temp_program = heapq.heappop(min_heap)
            heapq.heappush(max_heap, (-temp_program.profit, temp_program))
        if len(max_heap) == 0:
            return W
        W += (-heapq.heappop(max_heap)[0])

    return W

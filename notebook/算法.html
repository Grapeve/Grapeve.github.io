2.全排列

def next_permutation(strs: list, use: set, path: str, ans: list):
    if len(use) == len(strs):
        ans.append(path)
    else:
        for i in range(len(strs)):
            if i not in use:
                use.add(i)
                next_permutation(strs, use, path + strs[i], ans)
                use.remove(i)

3.给定一个字符串数组，对数组中的字符串按字典序进行排列，使排列后的数组中字符串组成的字符串最小。

eg: ["b", "ba", "a"] => abab

python
# 贪心
from functools import cmp_to_key
def cmp(x, y):
    if (x+y) > (x+y):
        return 1
    elif (x+y) < (x+y):
        return -1
    else:
        return 0

def leastStr(strs: list[str]) -> str:
    strs.sort(key=cmp_to_key(cmp))  # sort本身升序
    ans = ''.join(strs)
    return ans

# road[index...],自由选择放灯还是不放灯
# road[...index-1],已经放完灯了，放灯的位置存在lights里
# 要求选出能照亮所有.的方案，并且在这些方案中，返回最少需要几个灯。
def process(road:list[str], index:int, lights:set):
    if index == len(lights):
        for i in range(len(road)):
            if road[i] == '.':
                if i not in lights and (i-1) not in lights and (i+1) not in lights:
                    return float("inf")
        return len(lights)
    else:
        no = process(road, index + 1, lights)
        yes = float("inf")
        if road[index] == '.':
            lights.add(index)
            yes = process(road, index+1, lights)
            lights.remove(index)   # 恢复现场lights
        return min(no, yes)

import heapq
def less_money_split_gold(arr:list[int]):
    heap = []
    for num in arr:
        heapq.heappush(heap,num)
    sum = 0
    cur = 0
    while len(heap) > 1:
        cur = heapq.heappop(heap) + heapq.heappop(heap)
        sum += cur
        heapq.heappush(heap,cur)
    return sum

7.每做完一个项目，马上获得利益，并可以去做下一个项目。不并行做项目。问做K个项目能获得的最大钱数。Profits表示利润，Capital表示做项目需要的资金，W是初始资金。

python
class Program:
    def __init__(self, profit, capital):
        self.profit = profit
        self.capital = capital

    def __lt__(self, program):
        return self.capital < program.capital


def find_max_capital(K: int, W: int, Profits: list[int], Capital: list[int]):
    programs = []
    for i in range(len(Profits)):
        programs.append(Program(Profits[i], Capital[i]))

    min_heap: list[Program] = []
    max_heap: list[Program] = []

    for program in programs:
        heapq.heappush(min_heap, program)  # 小根堆根据capital

    for i in range(K):
        while len(min_heap) != 0 and min_heap[0].capital <= W:
            temp_program = heapq.heappop(min_heap)
            heapq.heappush(max_heap, (-temp_program.profit, temp_program))
        if len(max_heap) == 0:
            return W
        W += (-heapq.heappop(max_heap)[0])

    return W
